\section{Introduction}
% ══════════════════════════════════════════════════════════════════════════

We study the estimation of the Population Average Treatment Effect
(PATE) in A/B tests where some users are ``new'' and lack a
pre-experiment outcome~$\Ypre$.  The CUPED estimator
\citep{deng2013improving} uses $\Ypre$ to reduce variance, but is
undefined for new users.  We propose a \emph{stratified} approach:
estimate the treatment effect separately on returning and new users,
then combine the stratum-level estimates with proper weighting and
standard errors.

% ══════════════════════════════════════════════════════════════════════════
\section{Data Generating Process}
\label{sec:dgp}
% ══════════════════════════════════════════════════════════════════════════

For user $i \in \{1,\dots,n\}$:

\begin{enumerate}
\item Treatment assignment: $D_i \iid \mathrm{Bernoulli}(p)$.
\item New-user indicator: $N_i \iid \mathrm{Bernoulli}(\pi)$, independent of $D_i$.
\item Pre-period outcome (latent for all users):
      $\Ypre^{*}_i \iid \mathcal{N}(\mu_0, \sigma_0^2)$.
\item Observed pre-period outcome:
      $\Ypre_i = \begin{cases} \Ypre^{*}_i & \text{if } N_i = 0 \text{ (returning)}, \\
                                \text{missing} & \text{if } N_i = 1 \text{ (new)}.\end{cases}$
\item Individual treatment effect:
      $\tau_i = \tau + \sigma_h z_i, \quad z_i \iid \mathcal{N}(0,1)$.
      When $\sigma_h = 0$ the effect is homogeneous.
\item Post-period outcome:
      $Y_i = \beta_0 \Ypre^{*}_i + \tau_i D_i + \varepsilon_i, \quad
       \varepsilon_i \iid \mathcal{N}(0, \sigma_\varepsilon^2)$.
\end{enumerate}

The PATE is $\tau = \E[\tau_i]$.
The approximate correlation between $\Ypre^{*}$ and $Y$ (under $D=0$) is
\[
  \rho = \frac{\beta_0 \sigma_0^2}
              {\sqrt{\sigma_0^2\bigl(\beta_0^2 \sigma_0^2 + \sigma_\varepsilon^2\bigr)}}\,.
\]

% ══════════════════════════════════════════════════════════════════════════
\section{Estimators}
\label{sec:estimators}
% ══════════════════════════════════════════════════════════════════════════

For every estimator~$k$ we report five quantities:
\[
  \tauhat_k, \qquad
  \SE_k, \qquad
  \bigl[\tauhat_k \pm 1.96\,\SE_k\bigr], \qquad
  t_k = \frac{\tauhat_k}{\SE_k}, \qquad
  p_k = 2\,\Phi\!\bigl(-|t_k|\bigr),
\]
where $\Phi$ is the standard normal CDF.
We reject $H_0\!: \tau = 0$ at level $\alpha=0.05$ when $p_k < 0.05$.

% ── 3.1 Naive ─────────────────────────────────────────────────────────────
\subsection{Naive difference-in-means}

\[
  \tauhat_{\mathrm{naive}}
  = \bar{Y}_T - \bar{Y}_C,
  \qquad
  \SE(\tauhat_{\mathrm{naive}})
  = \sqrt{\frac{s_T^2}{n_T} + \frac{s_C^2}{n_C}}\,,
\]
where $s_T^2, s_C^2$ are sample variances within treatment and control.

% ── 3.2 CUPED ─────────────────────────────────────────────────────────────
\subsection{CUPED (returning users)}
\label{sec:cuped}

On users with observed $\Ypre$, we fit by OLS:
\begin{equation}\label{eq:cuped}
  Y_i = \alpha + \beta D_i + \mu \Ypre_i + e_i\,.
\end{equation}
The treatment-effect estimate is $\betahat$, the OLS coefficient on $D_i$.
Its standard error is the usual homoskedastic OLS SE:
\[
  \SE(\betahat) = \sqrt{\hat\sigma_e^2 \, [(X'X)^{-1}]_{22}},
  \qquad
  \hat\sigma_e^2 = \frac{\sum_i \hat{e}_i^2}{n_\bR - 3}\,.
\]

% ── 3.3 Stratified estimation ─────────────────────────────────────────────
\subsection{Stratified estimation with new users}
\label{sec:stratified}

When a fraction $\pi > 0$ of users are new, we estimate the treatment
effect separately in each stratum:

\begin{align*}
  \text{Returning} \; (\bR): & \quad
    \betahat_\bR,\; \SE_\bR
    \quad\text{from OLS CUPED \eqref{eq:cuped} on } n_\bR \text{ users}, \\
  \text{New} \; (\bN): & \quad
    \tauhat_\bN = \bar Y_{T,\bN} - \bar Y_{C,\bN},\;
    \SE_\bN
    \quad\text{from Welch diff-in-means on } n_\bN \text{ users}.
\end{align*}

\paragraph{Population-weighted PATE.}
With weights $w_\bR = n_\bR/n$, $w_\bN = n_\bN/n$:
\begin{align}
  \tauhat_{\mathrm{pop}}
  &= w_\bR\,\betahat_\bR + w_\bN\,\tauhat_\bN\,,
  \label{eq:tau_pop} \\
  \Var(\tauhat_{\mathrm{pop}})
  &= w_\bR^2\,\SE_\bR^2 + w_\bN^2\,\SE_\bN^2\,,
  \label{eq:var_pop}
\end{align}
where independence of the two strata (disjoint users) justifies the
variance decomposition.  This estimator targets the overall PATE and
weights each stratum by its population share.

\paragraph{Precision-weighted (minimum-variance).}
The inverse-variance-weighted combination:
\begin{align}
  \tauhat_{\mathrm{prec}}
  &= \frac{
      \betahat_\bR / V_\bR + \tauhat_\bN / V_\bN
    }{
      1/V_\bR + 1/V_\bN
    }\,,
  \qquad V_k = \SE_k^2\,,
  \label{eq:tau_prec} \\
  \Var(\tauhat_{\mathrm{prec}})
  &= \frac{1}{1/V_\bR + 1/V_\bN}\,.
  \label{eq:var_prec}
\end{align}
Under homogeneous treatment effects ($\sigma_h = 0$), both strata
target the same~$\tau$ and precision-weighting is efficient.  Under
heterogeneity, precision-weighting may deviate from PATE because it
up-weights the lower-variance stratum (returning users with CUPED).

% ── 3.4 Bayesian ──────────────────────────────────────────────────────────
\subsection{Bayesian normal-normal estimator}

Prior: $\tau \sim \mathcal{N}(m_0, v_0)$.
Likelihood approximation: $\tauhat_{\mathrm{naive}} \mid \tau \sim \mathcal{N}(\tau, \hat V)$
where $\hat V = s_T^2/n_T + s_C^2/n_C$.
Posterior:
\begin{align}
  \tau \mid \text{data}
  &\sim \mathcal{N}(\mu_{\mathrm{post}}, \sigma_{\mathrm{post}}^2)\,,
  \label{eq:bayes} \\
  \sigma_{\mathrm{post}}^2
  &= \Bigl(\frac{1}{v_0} + \frac{1}{\hat V}\Bigr)^{-1},
  \qquad
  \mu_{\mathrm{post}}
  = \sigma_{\mathrm{post}}^2
    \Bigl(\frac{m_0}{v_0} + \frac{\tauhat_{\mathrm{naive}}}{\hat V}\Bigr).
  \nonumber
\end{align}
We report $\mu_{\mathrm{post}}$ as the point estimate and
$\sigma_{\mathrm{post}}$ as the SE (posterior SD).

% ══════════════════════════════════════════════════════════════════════════
\section{Single-Experiment Analysis (Primary Use Case)}
\label{sec:single}
% ══════════════════════════════════════════════════════════════════════════

The primary use case is analyzing a \emph{single} completed experiment.
Given data $\{(D_i,\, \Ypre_i,\, Y_i)\}_{i=1}^n$, the analyst asks
three questions, each requiring a point estimate, standard error,
confidence interval, and $p$-value.

% ── 4.1 Existing users ───────────────────────────────────────────────────
\subsection{ATE on existing (returning) users}

Restrict to the $n_\bR$ users with observed~$\Ypre$.
Fit the CUPED model by OLS:
\[
  Y_i = \alpha + \beta D_i + \mu \Ypre_i + e_i,
  \qquad i \in \bR.
\]
The estimator $\betahat_\bR$ and its standard error
$\SE_\bR = \sqrt{\hat\sigma_e^2 [(X'X)^{-1}]_{22}}$ yield:
\begin{align*}
  \text{95\% CI}_\bR &= \bigl[\betahat_\bR - 1.96\,\SE_\bR,\;\;
                               \betahat_\bR + 1.96\,\SE_\bR\bigr], \\
  t_\bR &= \betahat_\bR / \SE_\bR, \\
  p_\bR &= 2\,\Phi(-|t_\bR|).
\end{align*}
The null hypothesis is $H_0\!: \beta = 0$ (no treatment effect among
returning users).

% ── 4.2 New users ────────────────────────────────────────────────────────
\subsection{ATE on new users}

Restrict to the $n_\bN$ users with missing~$\Ypre$.
No covariate adjustment is possible, so we use the difference-in-means:
\[
  \tauhat_\bN = \bar Y_{T,\bN} - \bar Y_{C,\bN},
  \qquad
  \SE_\bN = \sqrt{\frac{s_{T,\bN}^2}{n_{T,\bN}} +
                  \frac{s_{C,\bN}^2}{n_{C,\bN}}}\,.
\]
\begin{align*}
  \text{95\% CI}_\bN &= \bigl[\tauhat_\bN \pm 1.96\,\SE_\bN\bigr], \\
  t_\bN &= \tauhat_\bN / \SE_\bN, \\
  p_\bN &= 2\,\Phi(-|t_\bN|).
\end{align*}
The null hypothesis is $H_0\!: \tau_\bN = 0$.

% ── 4.3 Overall ATE ─────────────────────────────────────────────────────
\subsection{Overall ATE}

The population-weighted estimator \eqref{eq:tau_pop} combines the two
strata:
\[
  \tauhat_{\mathrm{pop}} = w_\bR\,\betahat_\bR + w_\bN\,\tauhat_\bN,
  \qquad w_\bR = n_\bR/n,\; w_\bN = n_\bN/n.
\]
Because the strata are disjoint (independent), the standard error is
\[
  \SE_{\mathrm{pop}} = \sqrt{w_\bR^2\,\SE_\bR^2 + w_\bN^2\,\SE_\bN^2}\,.
\]
\begin{align*}
  \text{95\% CI}_{\mathrm{pop}} &= \bigl[\tauhat_{\mathrm{pop}}
    \pm 1.96\,\SE_{\mathrm{pop}}\bigr], \\
  t_{\mathrm{pop}} &= \tauhat_{\mathrm{pop}} / \SE_{\mathrm{pop}}, \\
  p_{\mathrm{pop}} &= 2\,\Phi(-|t_{\mathrm{pop}}|).
\end{align*}

\medskip
\noindent
\textbf{Key observations:}
\begin{itemize}
  \item The SE for existing users is smaller than the naive SE because
        CUPED uses $\Ypre$ to absorb variation.
  \item The SE for new users is larger (fewer users, no covariate
        adjustment).
  \item The population-weighted overall SE is between the two, reflecting
        the variance from both strata.
  \item The precision-weighted SE is the smallest achievable combined SE,
        but the estimand may differ from PATE under heterogeneity.
\end{itemize}

% ── 4.5 Testing equality ────────────────────────────────────────────────
\subsection{Testing equality of effects across strata}

To test whether returning and new users respond differently to
treatment ($H_0\!: \beta_\bR = \tau_\bN$), compute
\[
  t_{\mathrm{diff}} = \frac{\betahat_\bR - \tauhat_\bN}
                           {\sqrt{\SE_\bR^2 + \SE_\bN^2}}\,,
  \qquad
  p_{\mathrm{diff}} = 2\,\Phi\!\bigl(-|t_{\mathrm{diff}}|\bigr).
\]
Independence of the two strata justifies adding the variances.
Rejecting this null at $\alpha=0.05$ would indicate heterogeneity
between new and returning users.

\section{Monte Carlo Design}
\label{sec:mc}
% ══════════════════════════════════════════════════════════════════════════

We run $S = 1{,}000$ independent replications.  In each replication~$s$:
\begin{enumerate}
  \item Draw data $\{(D_i, \Ypre_i, Y_i, N_i)\}_{i=1}^n$ from the DGP
        in \Cref{sec:dgp}.
  \item Compute each estimator $k$: $(\tauhat_k^{(s)},\;\SE_k^{(s)})$.
  \item Record whether the 95\% CI
        $[\tauhat_k^{(s)} \pm 1.96\,\SE_k^{(s)}]$ covers $\tau$.
\end{enumerate}

\paragraph{Performance metrics} (across $S$ simulations for estimator $k$):

\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{ll}
\toprule
Metric & Definition \\
\midrule
Bias
  & $\displaystyle\frac{1}{S}\sum_{s=1}^S \tauhat_k^{(s)} - \tau$ \\[4pt]
Empirical SD
  & $\displaystyle\mathrm{sd}\bigl(\{\tauhat_k^{(s)}\}_{s=1}^S\bigr)$ \\[4pt]
Mean SE
  & $\displaystyle\frac{1}{S}\sum_{s=1}^S \SE_k^{(s)}$ \\[4pt]
SE/SD Ratio
  & $\displaystyle\text{Mean SE} \;/\; \text{Empirical SD}$
    \quad(should $\approx 1$ if SE is calibrated) \\[4pt]
RMSE
  & $\displaystyle\sqrt{\frac{1}{S}\sum_{s=1}^S (\tauhat_k^{(s)} - \tau)^2}$ \\[4pt]
Coverage
  & Fraction of replications where $\tau \in [\tauhat_k^{(s)} \pm 1.96\,\SE_k^{(s)}]$
    \quad(should $\approx 0.95$) \\[4pt]
Power
  & Fraction of replications where $0 \notin [\tauhat_k^{(s)} \pm 1.96\,\SE_k^{(s)}]$ \\
\bottomrule
\end{tabular}
\end{center}

At baseline, $\rho \approx 0.37$.  Increasing $\beta_0$ or decreasing
$\sigma_\varepsilon$ raises $\rho$ and amplifies CUPED's variance
reduction.  Increasing $\pi$ shifts more weight to the new-user
stratum, where no variance reduction is available.

% ══════════════════════════════════════════════════════════════════════════
\section{Scenarios}
\label{sec:scenarios}
% ══════════════════════════════════════════════════════════════════════════

We vary one parameter at a time from the baseline:

\begin{enumerate}
  \item \textbf{No new users} ($\pi=0$): CUPED operates on the full
        sample. Stratified and plain CUPED coincide.
  \item \textbf{Low new-user fraction} ($\pi=0.1$): Returning stratum
        dominates; precision-weighting close to population-weighting.
  \item \textbf{High new-user fraction} ($\pi=0.5$): Large new-user
        stratum; population-weighted estimate substantially different
        from returning-only CUPED.
  \item \textbf{Strong autocorrelation} ($\beta_0=1.5$): CUPED's
        variance reduction is large; precision-weighting leans heavily
        on returning users.
  \item \textbf{Weak autocorrelation} ($\beta_0=0.1$): CUPED barely
        helps; all estimators behave similarly.
  \item \textbf{Heterogeneous effects} ($\sigma_h=0.1$): Precision-weighting
        may deviate from PATE because it over-weights the lower-variance
        stratum.
\end{enumerate}

% ══════════════════════════════════════════════════════════════════════════
\section{Expected Results}
% ══════════════════════════════════════════════════════════════════════════

\begin{itemize}
  \item All estimators should be approximately unbiased for PATE when
        $\sigma_h = 0$.
  \item CUPED (returning) should have lower RMSE than naive when
        $\rho > 0$.
  \item Population-weighted PATE should have correct coverage ($\approx 0.95$)
        and SE/SD ratio $\approx 1$.
  \item Precision-weighted should have lower RMSE than population-weighted
        (by construction), but may exhibit slight bias for PATE under
        heterogeneity if the strata have different conditional ATEs.
  \item The Bayesian estimator with an informative prior ($\sqrt{v_0}=0.1$)
        will exhibit shrinkage toward $m_0$; coverage may drop below 0.95
        if the prior is misspecified.
\end{itemize}
